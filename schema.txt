CREATE TABLE public.allianceState (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  myAllianceName text NOT NULL,
  opponentAllianceId text,
  isInWar boolean DEFAULT false,
  isPopulated boolean DEFAULT false,
  lastUpdated timestamp with time zone DEFAULT now(),
  warStartTime timestamp with time zone,
  ally_regen smallint DEFAULT 3,
  enemy_regen smallint DEFAULT 3,
  CONSTRAINT allianceState_pkey PRIMARY KEY (id)
);
CREATE TABLE public.allianceStatus (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  allianceId text NOT NULL,
  allianceName text,
  isAlly boolean,
  initialWarPoints bigint DEFAULT 0,
  currentWarPoints bigint DEFAULT 0,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT allianceStatus_pkey PRIMARY KEY (id)
);
CREATE TABLE public.memberStatus (
  id bigint GENERATED ALWAYS AS IDENTITY NOT NULL,
  memberId text NOT NULL,
  memberName text,
  isAlly boolean,
  initialWarPoints bigint DEFAULT 0,
  currentWarPoints bigint DEFAULT 0,
  CONSTRAINT memberStatus_pkey PRIMARY KEY (id)
);
CREATE TABLE public.planet_timers (
  id integer NOT NULL DEFAULT nextval('planet_timers_id_seq'::regclass),
  account_id integer NOT NULL,
  planet_index integer NOT NULL,
  dead_until timestamp with time zone NOT NULL,
  created_at timestamp with time zone DEFAULT now(),
  CONSTRAINT planet_timers_pkey PRIMARY KEY (id)
);
CREATE TABLE public.planets (
  id integer GENERATED ALWAYS AS IDENTITY NOT NULL UNIQUE,
  accountId integer,
  name text,
  planetId smallint,
